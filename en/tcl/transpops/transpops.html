::transpopsTop

The transpops package displays popup messages.

The messages are read from a file and after pressing a hotkey are shown one by one under the mouse pointer, in transparenting mode.

Lisense: MIT.

Runs as follows:


package require transpops
::transpops::run transpops.txt <Control-q> .win

where:

  • transpops.txt - text file containing messages
  • Control-q - event to start the popup of messages
  • .win - toplevel window the event is bound to

proc ::transpops::run {fname event win {fg1 #000000} {bg1 #FBFB95}} {

  # Initializes and runs the popup messages.
  #   fname - name of file containing the messages
  #   event - event bound to the window to run popups
  #   win - the window's path
  #   fg1 - foreground of popups
  #   bg1 - background of popups

  set ::transpops::my::fg $fg1
  set ::transpops::my::bg $bg1
  set chan [open $fname]
  chan configure $chan -encoding utf-8
  set ::transpops::my::msgs [split [read $chan] \n]
  close $chan
  set ::transpops::my::imsgs 0
  bind $win $event ::transpops::my::Show
}

proc ::transpops::my::Popup {msg} {

  # Popups a message in the popup window.
  #   msg - the message

  variable wmsgs
  variable alpha
  variable alphaincr
  variable cntwait
  if {![winfo exists $wmsgs]} return
  if {$cntwait>0} {
    incr cntwait -1
  } else {
    set alpha [expr {$alpha+$alphaincr}]
    if {$alpha>1.0} {
      set cntwait [expr {int(max(2,[string length $msg]/12.)/$alphaincr)}]
      set alphaincr [expr {-$alphaincr}]
      set alpha 1.0
    } elseif {$alpha<0.0} {
      catch {destroy $wmsgs}
      return
    }
    catch {wm attributes $wmsgs -alpha $alpha}
  }
  after 10 "::transpops::my::Popup {$msg}"
}

proc ::transpops::my::Show {} {

  # Makes the popup toplevel window and starts popups.

  variable msgs
  variable imsgs
  variable wmsgs
  variable alpha
  variable alphaincr
  variable fg
  variable bg
  catch {destroy $wmsgs}
  if {[incr imsgs]>=[llength $msgs]} return
  set msg [string map {\\n \n} [lindex $msgs $imsgs-1]]
  toplevel $wmsgs
  wm withdraw $wmsgs
  wm overrideredirect $wmsgs 1
  label $wmsgs.l -padx 30 -pady 30 -foreground $fg -background $bg  -font {-weight bold -size 20 -family Quicksand} -relief solid -text $msg -justify left
  set alpha 0.0
  set alphaincr [expr {abs($alphaincr)}]
  pack $wmsgs.l -fill both -expand true
  set alpha 0.0
  set alphaincr 0.007
  bind $wmsgs <ButtonPress> {set ::transpops::my::cntwait 0}
  set x [expr {[winfo pointerx .]+10}]
  set y [expr {[winfo pointery .]+10}]
  wm geometry $wmsgs +$x+$y
  after 10 "wm deiconify $wmsgs; wm attributes $wmsgs -alpha 0.0 -topmost 1"
  after 10 "::transpops::my::Popup {$msg}"
}
